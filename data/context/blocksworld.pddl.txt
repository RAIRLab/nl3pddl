(define (domain blocks)

    (:requirements :strips :typing)

    (:types block)

    (:predicates 
        (on ?x - block ?y - block)
        (ontable ?x - block)
        (clear ?x - block)
        (handempty)
        (holding ?x - block)
    )

    (:action pick-up
        :parameters (?x - block)
        :precondition ()
        :effect ()
    )

    (:action put-down
        :parameters (?x - block)
        :precondition ()
        :effect ()
    )

    (:action stack
        :parameters (?x - block ?y - block)
        :precondition ()
        :effect ()
    )

    (:action unstack
        :parameters (?x - block ?y - block)
        :precondition ()
        :effect ()
    )
)
